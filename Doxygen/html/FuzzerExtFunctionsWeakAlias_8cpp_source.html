21 #define EXT_FUNC(NAME, RETURN_TYPE, FUNC_SIG, WARN) \

33 template <typename T>
34 static T *GetFnPtr(T *Fun, T *FunDef, const char *FnName, bool WarnIfMissing) {
35  if (Fun == FunDef) {
36  if (WarnIfMissing)
37  Printf("WARNING: Failed to find function \"%s\".\n", FnName);
38  return nullptr;
39  }
40  return Fun;
41 }
